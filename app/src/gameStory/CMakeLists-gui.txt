# CMakeLists.txt for SFML GUI app
cmake_minimum_required(VERSION 3.10)
project(gameStoryGUI)
set(CMAKE_CXX_STANDARD 17)

find_package(SFML 2.5 COMPONENTS graphics window system audio REQUIRED)

add_executable(gameStoryGUI main.cpp)
target_link_libraries(gameStoryGUI sfml-graphics sfml-window sfml-system sfml-audio)
