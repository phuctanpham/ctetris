\documentclass[12pt,a4paper]{article}

\IfFileExists{src/libs/body.tex}{\input{src/libs/body.tex}}{\input{../libs/body.tex}}
\usepackage{docmute}
\hypersetup{pdftitle={Báo cáo Kỹ năng nghề nghiệp - Mục lục}}

\newcommand{\NoTocNumberLine}[1]{}

\makeatletter
\@ifundefined{babel@toc}{\def\babel@toc#1#2{}}{}
\makeatother

\newcommand{\PersistMainTOC}{%
	\immediate\write18{cp -f \jobname.toc src/index-main.toc >/dev/null 2>&1 || true}%
	\immediate\write18{cp -f \jobname.toc index-main.toc >/dev/null 2>&1 || true}%
}

\newcommand{\RenderMainTOC}{%
    \begingroup
        \let\numberline\NoTocNumberLine
        \tableofcontents
    \endgroup
	\PersistMainTOC
}

\newcommand{\RenderStandaloneTOC}{%
	\section*{\contentsname}%
	\begingroup
		\let\numberline\NoTocNumberLine
		\IfFileExists{index-main.toc}{%
			\makeatletter\input{index-main.toc}\makeatother%
		}{%
			\IfFileExists{src/index-main.toc}{%
				\makeatletter\input{src/index-main.toc}\makeatother%
			}{%
				\IfFileExists{../src/index-main.toc}{%
					\makeatletter\input{../src/index-main.toc}\makeatother%
				}{%
					\IfFileExists{../main.toc}{%
						\makeatletter\input{../main.toc}\makeatother%
					}{%
						\IfFileExists{../output.toc}{%
							\makeatletter\input{../output.toc}\makeatother%
						}{%
							\IfFileExists{main.toc}{%
								\makeatletter\input{main.toc}\makeatother%
							}{%
								\IfFileExists{output.toc}{%
									\makeatletter\input{output.toc}\makeatother%
								}{%
									\tableofcontents
								}%
							}%
						}%
					}%
				}%
			}%
		}{%
			\tableofcontents
		}%
	\endgroup
}

\begin{document}
\pagenumbering{arabic}
\ifdefined\MainDocumentBuildFlag
	\RenderMainTOC
\else
	\RenderStandaloneTOC
\fi
\clearpage
\end{document}
